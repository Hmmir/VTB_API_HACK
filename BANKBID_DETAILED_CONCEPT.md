class AutoBiddingEngine:
    def should_bid(self, request: Request, bank: Bank) -> bool:
        """Decide if bank should participate in auction"""
        rules = bank.bidding_rules
        
        # Check filters
        if request.amount < rules.min_amount:
            return False
        if request.amount > rules.max_amount:
            return False
        if request.credit_score < rules.min_credit_score:
            return False
            
        # Check capacity
        if bank.daily_budget_spent >= bank.daily_budget:
            return False
            
        return True
    
    def calculate_bid(self, request: Request, 
                      competitors: List[Bid]) -> Decimal:
        """Calculate optimal bid rate"""
        base_rate = self.get_base_rate(request)
        
        if not competitors:
            return base_rate + Decimal('0.5')  # Premium
        
        best_competitor = min(competitors, key=lambda b: b.rate)
        our_competitive_rate = best_competitor.rate + Decimal('0.1')
        
        # Don't go below profitability
        if our_competitive_rate < base_rate:
            return None  # Skip this auction
            
        return our_competitive_rate
